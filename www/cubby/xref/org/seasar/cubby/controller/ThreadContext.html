16  package org.seasar.cubby.controller;
17  
18  import static org.seasar.cubby.CubbyConstants.RES_MESSAGES;
19  
20  import java.util.Locale;
21  import java.util.Map;
22  import java.util.ResourceBundle;
23  
24  import javax.servlet.http.HttpServletRequest;
25  
26  import org.seasar.framework.container.SingletonS2Container;
27  import org.seasar.framework.util.ResourceBundleUtil;
28  
29  /**
30   * 実行スレッドのコンテキスト情報です。
31   * 
32   * @author baba
33   */
34  public class ThreadContext {
35  
36  	/** ThreadContext を保存するスレッドローカル。 */
37  	private static final ThreadLocal<ThreadContext> CONTEXT = new ThreadLocal<ThreadContext>() {
38  
39  		@Override
40  		protected ThreadContext initialValue() {
41  			return new ThreadContext();
42  		}
43  
44  	};
45  
46  	/**
47  	 * スレッドローカルから現在の実行スレッドに関する情報を削除します。
48  	 */
49  	public static void remove() {
50  		CONTEXT.remove();
51  	}
52  
53  	/**
54  	 * 現在の実行スレッドに関連付けられたリクエストを取得します。
55  	 * 
56  	 * @return リクエスト
57  	 */
58  	public static HttpServletRequest getRequest() {
59  		return CONTEXT.get().request;
60  	}
61  
62  	/**
63  	 * 現在の実行スレッドに指定されたリクエストを関連付けます。
64  	 * 
65  	 * @param request
66  	 *            リクエスト
67  	 */
68  	public static void setRequest(final HttpServletRequest request) {
69  		CONTEXT.get().request = request;
70  	}
71  
72  	/**
73  	 * Cubby の全体的な設定情報を取得します。
74  	 * 
75  	 * @return Cubby の全体的な設定情報
76  	 */
77  	public static CubbyConfiguration getConfiguration() {
78  		return SingletonS2Container.getComponent(CubbyConfiguration.class);
79  	}
80  
81  	/**
82  	 * 現在の実行スレッドに関連付けられたリクエストに対応するメッセージ用の {@link ResourceBundle} を取得します。
83  	 * 
84  	 * @return リソースバンドル
85  	 */
86  	public static ResourceBundle getMessagesResourceBundle() {
87  		final ThreadContext context = CONTEXT.get();
88  		if (context.resourceBundle == null) {
89  			final Locale locale;
90  			if (context.request == null) {
91  				locale = Locale.getDefault();
92  			} else {
93  				locale = context.request.getLocale();
94  			}
95  			context.resourceBundle = ResourceBundleUtil.getBundle(RES_MESSAGES,
96  					locale);
97  		}
98  		return context.resourceBundle;
99  	}
100 
101 	/**
102 	 * {@link #getMessagesResourceBundle()} で取得できる {@link ResourceBundle} を変換した
103 	 * {@link Map} を取得します。
104 	 * 
105 	 * @return メッセージの {@link Map}
106 	 */
107 	public static Map<?, ?> getMessagesMap() {
108 		final ThreadContext context = CONTEXT.get();
109 		if (context.messages == null) {
110 			final ResourceBundle resourceBundle = getMessagesResourceBundle();
111 			context.messages = ResourceBundleUtil.convertMap(resourceBundle);
112 		}
113 		return context.messages;
114 	}
115 
116 	/**
117 	 * インスタンス化します。
118 	 */
119 	private ThreadContext() {
120 	}
121 
122 	/** リクエスト。 */
123 	private HttpServletRequest request;
124 
125 	/** リソースバンドル。 */
126 	private ResourceBundle resourceBundle = null;
127 
128 	/** メッセージの {@link Map} */
129 	private Map<?, ?> messages = null;
130 
131 }
