8   /**
9    * アクションメソッドの URL。
10   * <p>
11   * アクションメソッドを起動するための URL を指定するアノテーションです。
12   * </p>
13   * <p>
14   * 使用例
15   * 
16   * <pre>
17   * &#064;Url(&quot;fuga&quot;)
18   * public class HogeAction {
19   * 	// -&gt; &quot;/fuga/m1&quot;
20   * 	public ActionResult m1() {
21   * 	}
22   * 
23   * 	&#064;Url(&quot;list&quot;)
24   * 	// -&gt; &quot;/fuga/list&quot;
25   * 	public ActionResult m2() {
26   * 	}
27   * 
28   * 	&#064;Url(&quot;/xxx/yyy&quot;)
29   * 	// -&gt; &quot;/xxx/yyy&quot;
30   * 	public ActionResult m3() {
31   * 	}
32   * 
33   * 	&#064;Url(&quot;/{id}/edit&quot;)
34   * 	// {id}部分をリクエストパラメータに追加
35   * 	public ActionResult m4() {
36   * 	}
37   * 
38   * 	&#064;Url(&quot;/{userId,a-z}/edit&quot;)
39   * 	// {userId}部分をリクエストパラメータに追加。ユーザID部分は小文字アルファベットのみ。
40   * 	public ActionResult m5() {
41   * 	}
42   * }
43   * </pre>
44   * 
45   * </p>
46   * 
47   * @author agata
48   * @author baba
49   * @since 1.0
50   */
51  @Retention(RetentionPolicy.RUNTIME)
52  @Target( { ElementType.METHOD, ElementType.TYPE })
53  public @interface Url {
54  
55  	/**
56  	 * アクションメソッドを起動する対象となる HTTP メソッド。
57  	 * 
58  	 * @author agata
59  	 * @author baba
60  	 * @since 1.0
61  	 */
62  	enum RequestMethod {
63  		/** HTTP GET */
64  		GET,
65  		/** HTTP HEAD */
66  		HEAD,
67  		/** HTTP POST */
68  		POST,
69  		/** HTTP PUT */
70  		PUT,
71  		/** HTTP DELETE */
72  		DELETE,
73  		/** HTTP OPTIONS */
74  		OPTIONS,
75  		/** HTTP TRACE */
76  		TRACE,
77  	}
78  
79  	/**
80  	 * アクションメソッドのバインディング用URLを指定します。
81  	 * <p>
82  	 * URLはアクションクラスのURL＋アクションメソッドのURLで決定されます。
83  	 * ただし、先頭が『/』の場合コンテキストルートからの絶対パスとして解釈されます。
84  	 * </p>
85  	 * <p>
86  	 * {パラメータ名,正規表現}でプレースホルダーの指定ができます。
87  	 * </p>
88  	 * <p>
89  	 * 正規表現にマッチした場合、マッチした箇所が指定されたパラメータ名に追加され、アクションメソッドが実行されます。
90  	 * 正規表現は省略可能です。省略した場合「0-9a-zA-Z」と同じ意味になります。
91  	 * </p>
92  	 * 
93  	 * @return アクションメソッドのバインディング用URL
94  	 */
95  	String value() default "";
96  
97  	/**
98  	 * アクションメソッドが対応するHTTPのメソッドを指定します。
99  	 * 
100 	 * @return アクションメソッドが対応するHTTPのメソッド
101 	 */
102 	RequestMethod[] accept() default { RequestMethod.GET,
103 			RequestMethod.POST };
104 
105 }
