16  package org.seasar.cubby.aop;

18  import java.io.Serializable;
19  import java.lang.reflect.Method;
20  import java.util.HashSet;
21  import java.util.Set;
22  import java.util.regex.Pattern;
23  
24  import org.seasar.cubby.util.CubbyUtils;
25  import org.seasar.framework.aop.Pointcut;
26  import org.seasar.framework.exception.EmptyRuntimeException;

28  /**
29   * アクションメソッドに適用される {@link Pointcut}。
30   * 
31   * @author baba
32   */
33  public class ActionMethodPointcutImpl implements Pointcut, Serializable {
34  
35  	private static final long serialVersionUID = -5701826062675617117L;
36  
37  	private String[] methodNames;
38  
39  	private Pattern[] patterns;
40  
41  	/**
42  	 * {@link ActionMethodPointcutImpl}を作成します。
43  	 * 
44  	 * @param targetClass
45  	 * @throws EmptyRuntimeException
46  	 */
47  	public ActionMethodPointcutImpl(final Class<?> targetClass)
48  			throws EmptyRuntimeException {
49  
50  		if (targetClass == null) {
51  			throw new EmptyRuntimeException("targetClass");
52  		}
53  		setMethodNames(getMethodNames(targetClass));
54  	}
55  
56  	/**
57  	 * {@link ActionMethodPointcutImpl}を作成します。
58  	 * 
59  	 * @param methodNames
60  	 * @throws EmptyRuntimeException
61  	 */
62  	public ActionMethodPointcutImpl(final String[] methodNames)
63  			throws EmptyRuntimeException {
64  
65  		if (methodNames == null || methodNames.length == 0) {
66  			throw new EmptyRuntimeException("methodNames");
67  		}
68  		setMethodNames(methodNames);
69  	}
70  
71  	/**
72  	 * {@inheritDoc}
73  	 */
74  	public boolean isApplied(final Method targetMethod) {
75  		if (!CubbyUtils.isActionMethod(targetMethod)) {
76  			return false;
77  		}
78  
79  		final String methodName = targetMethod.getName();
80  		for (int i = 0; i < patterns.length; ++i) {
81  			if (patterns[i].matcher(methodName).matches()) {
82  				return true;
83  			}
84  		}
85  		return false;
86  	}
87  
88  	/**
89  	 * 対象になったメソッド名の配列を返します。
90  	 * 
91  	 * @return 対象になったメソッド名の配列
92  	 */
93  	public String[] getMethodNames() {
94  		if (methodNames == null) {
95  			return null;
96  		}
97  		return methodNames.clone();
98  	}
99  
100 	private void setMethodNames(final String[] methodNames) {
101 		this.methodNames = methodNames;
102 		patterns = new Pattern[methodNames.length];
103 		for (int i = 0; i < patterns.length; ++i) {
104 			patterns[i] = Pattern.compile(methodNames[i]);
105 		}
106 	}
107 
108 	private static String[] getMethodNames(final Class<?> targetClass) {
109 		final Set<String> methodNameSet = new HashSet<String>();
110 		if (targetClass.isInterface()) {
111 			addInterfaceMethodNames(methodNameSet, targetClass);
112 		}
113 		for (Class<?> clazz = targetClass; clazz != Object.class
114 				&& clazz != null; clazz = clazz.getSuperclass()) {
115 			final Class<?>[] interfaces = clazz.getInterfaces();
116 			for (int i = 0; i < interfaces.length; ++i) {
117 				addInterfaceMethodNames(methodNameSet, interfaces[i]);
118 			}
119 		}
120 		return methodNameSet.toArray(new String[methodNameSet.size()]);
121 
122 	}
123 
124 	private static void addInterfaceMethodNames(
125 			final Set<String> methodNameSet, final Class<?> interfaceClass) {
126 		final Method[] methods = interfaceClass.getDeclaredMethods();
127 		for (int j = 0; j < methods.length; j++) {
128 			methodNameSet.add(methods[j].getName());
129 		}
130 		final Class<?>[] interfaces = interfaceClass.getInterfaces();
131 		for (int i = 0; i < interfaces.length; ++i) {
132 			addInterfaceMethodNames(methodNameSet, interfaces[i]);
133 		}
134 	}
