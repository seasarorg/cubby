37      private static final Random RANDOM = new Random();

39      /**
40       * ユニークなトークンを生成します。
41       * @return ユニークなトークン
42       */
43      public static String generateGUID() {
44          return new BigInteger(165, RANDOM).toString(36).toUpperCase();
45      }

47      /**
48       * トークン用のマップをセッションから取得します。
49       * <p>
50       * セッション中にトークン用のマップが存在しない場合、新規に生成します。
51       * トークン用のマップは{@link LruHashMap}を使い、トークンの保持上限付きのMapになります。
52       * </p>
53       * @param sessionMap
54       * @return トークン用のマップ
55       */
56  	@SuppressWarnings("unchecked")
57  	public
58  	static Map<String, String> getTokenMap(HttpSession session) {
59  		Map<String, String> tokenMap = (Map<String, String>) session.getAttribute(CubbyConstants.ATTR_TOKEN);
60  		if (tokenMap == null) {
61  			tokenMap = new LruHashMap(TOKEN_HISTORY_SIZE);
62  			session.setAttribute(ATTR_TOKEN, tokenMap);
63  		}
64  		return tokenMap;
65  	}
66  	
67  	/**
68  	 * トークンをセッション中のトークン用のMapにセットします。
69  	 * @param sessionMap セッション
70  	 * @param token トークン文字列
71  	 */
72  	public static void setToken(HttpSession session, String token) {
73  		Map<String, String> tokenMap = getTokenMap(session);
74  		synchronized (tokenMap) {
75  			tokenMap.put(token, null);
76  		}
77  	}
78  
79  	/**
80  	 * パラメータ中のトークン文字列とセッション中のトークン文字列を検証します。
81  	 * <p>
82  	 * セッション中に格納されたトークン用のMapのキーに、
83  	 * 指定されたトークン文字列が含まれるかどうかを判定し、Mapから取り除きます。
84  	 * </p>
85  	 * @param sessionMap セッション用のMap
86  	 * @param token トークン文字列
87  	 * @return 指定されたトークン文字列がセッション中に存在したら<code>true</code>、それ以外は<code>false</code>
88  	 */
89  	public static boolean validateToken(HttpSession session,
90  			String token) {
91  		Map<String, String> tokenMap = getTokenMap(session);
92  		synchronized (tokenMap) {
93  			boolean success = tokenMap.containsKey(token);
94  			tokenMap.remove(token);
95  			return success;
96  		}
97  	}
