74  	@Override
75  	public int doStartTag() throws JspException {
76  		final Map<String, String[]> outputValues = bindFormToOutputValues(this.value);
77  		pageContext.setAttribute(ATTR_OUTPUT_VALUES, outputValues,
78  				PageContext.PAGE_SCOPE);

80  		final JspWriter out = pageContext.getOut();
81  		try {

86  			throw new JspException(e);
87  		}
88  		return EVAL_BODY_INCLUDE;
89  	}

91  	private Map<String, String[]> bindFormToOutputValues(final Object value) {
92  		final Map<String, String[]> outputValues;
93  		if (value == null) {
94  			outputValues = Collections.emptyMap();
95  		} else {
96  			final ActionContext context = SingletonS2Container
97  					.getComponent(ActionContext.class);
98  			final FormDxo formDxo = context.getFormDxo();
99  			final Map<String, String[]> map = new HashMap<String, String[]>();
100 			formDxo.convert(value, map);
101 			outputValues = map;
102 		}
103 		return outputValues;
104 	}
