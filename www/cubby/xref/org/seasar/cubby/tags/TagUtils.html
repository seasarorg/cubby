133 	public static String toAttr(final Map<String, Object> map) {
134 		final StringBuilder sb = new StringBuilder();
135 		for (final Entry<String, Object> entry : map.entrySet()) {
136 			final String key = entry.getKey();
137 			if ("value".equals(key) || "checkedValue".equals(key)) {
138 				continue;
139 			}
140 			sb.append(key);
141 			sb.append("=\"");
142 			sb.append(CubbyFunctions.escapeHtml(entry.getValue()));
143 			sb.append("\" ");
144 		}

148 	public static boolean isChecked(final String value, final Object values) {
149 		if (values instanceof Collection) {
150 			return ((Collection<?>) values).contains(value);
151 		} else if (values.getClass().isArray()) {
152 			for (final Object v : (Object[]) values) {
153 				if (equalsValue(v, value)) {
154 					return true;
155 				}
156 			}
157 			return false;
158 		} else {
159 			return equalsValue(values, value);
160 		}
161 	}

163 	private static boolean equalsValue(final Object values, final Object value) {
164 		if (values == value) {
165 			return true;
166 		} else if (values == null) {
167 			return false;
168 		} else {
