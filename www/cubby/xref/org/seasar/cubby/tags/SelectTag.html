126 	@Override
127 	public void doTag() throws JspException, IOException {
128 		final JspContext context = this.getJspContext();
129 		final JspWriter out = context.getOut();
130 		final ActionErrors errors = errors(context);
131 		final Map<String, Object> dyn = this.getDynamicAttribute();
132 		final Map<String, String[]> outputValues = outputValues(context);
133 
134 		if (!errors.getFields().get(this.name).isEmpty()) {
135 			addClassName(dyn, "fieldError");
136 		}
137 
138 		final Object[] value = multipleFormValues(context, outputValues,
139 				this.name);
140 
141 		out.write("<select name=\"");
142 		out.write(this.name);
143 		out.write("\" ");
144 		out.write(toAttr(dyn));
145 		out.write(">\n");
146 
147 		if (emptyOption) {
148 			out.write("<option value=\"\">");
149 			out.write(CubbyFunctions.out(emptyOptionLabel));
150 			out.write("</option>\n");
151 		}
152 
153 		if (items.getClass().isArray()) {
154 			final OptionWriter optionWriter = new OptionWriter(
155 					new BeanItemAdaptor());
156 			for (final Object item : (Object[]) items) {
157 				optionWriter.write(out, item, value);
158 			}
159 		} else {
160 			final OptionWriter optionWriter;
161 			final Collection<?> collection;
162 			if (items instanceof Collection) {
163 				optionWriter = new OptionWriter(new BeanItemAdaptor());
164 				collection = (Collection<?>) items;
165 			} else if (items instanceof Map) {
166 				optionWriter = new OptionWriter(new EntryItemAdaptor());
167 				collection = ((Map<?, ?>) items).entrySet();
168 			} else {
169 				throw new JspTagException(MessageFormatter.getMessage(
170 						"ECUB1001", new Object[] { "items", items.getClass() }));
171 			}
172 			for (final Object item : collection) {
173 				optionWriter.write(out, item, value);
174 			}
175 		}
176 
177 		out.write("</select>\n");
178 	}

180 	static class OptionWriter {
181 
182 		private final ItemAdaptor itemAdaptor;
183 
184 		OptionWriter(final ItemAdaptor itemAdaptor) {
185 			this.itemAdaptor = itemAdaptor;
186 		}
187 
188 		public void write(final JspWriter out, final Object item,
189 				final Object value) throws IOException {
190 			out.write("<option value=\"");
191 			final String itemValue = DynamicAttributesTagSupport
192 					.toString(itemAdaptor.getItemValue(item));
193 			final String labelValue = DynamicAttributesTagSupport
194 					.toString(itemAdaptor.getLabelValue(item));
195 			out.write(CubbyFunctions.out(itemValue));
196 			out.write("\" ");
197 			out.write(selected(itemValue, value));
198 			out.write(">");
199 			out.write(CubbyFunctions.out(labelValue));
200 			out.write("</option>\n");
201 		}

203 		private String selected(final String value, final Object values) {
204 			if (value == null || values == null) {
205 				return "";
206 			}
207 			if (isChecked(value, values)) {
208 				return "selected=\"true\"";
209 			} else {
210 				return "";
211 			}
212 		}

223 	class BeanItemAdaptor implements ItemAdaptor {
224 
225 		public BeanItemAdaptor() throws JspTagException {
226 			if (valueProperty == null) {
227 				throw new JspTagException(MessageFormatter.getMessage(
228 						"ECUB1002", new Object[] { "items", "valueProperty" }));
229 			}
230 		}
231 
232 		public Object getItemValue(final Object item) {
233 			return property(item, valueProperty);
234 		}
235 
236 		public Object getLabelValue(final Object item) {
237 			final Object labelValue;
238 			if (labelProperty == null) {
239 				labelValue = getItemValue(item);
240 			} else {
241 				labelValue = property(item, labelProperty);
242 			}
243 			return labelValue;
244 		}
245 
246 		private Object property(final Object bean, final String propertyName) {
247 			final BeanDesc beanDesc = BeanDescFactory.getBeanDesc(bean
248 					.getClass());
249 			final PropertyDesc propertyDesc = beanDesc
250 					.getPropertyDesc(propertyName);
251 			return propertyDesc.getValue(bean);
252 		}
