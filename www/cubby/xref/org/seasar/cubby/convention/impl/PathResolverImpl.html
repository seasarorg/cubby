89  	private void add(final String regexp,
90  			final Class<? extends Action> actionClass, final Method method,
91  			final Map<Pattern, RoutingInfo> patternToRoutingInfoMap,
92  			final Url.RequestMethod... requestMethods) {
93  
94  		String actionFullName = regexp;
95  		final List<String> uriParameterNames = new ArrayList<String>();
96  		final Matcher matcher = routingPattern.matcher(actionFullName);
97  		while (matcher.find()) {
98  			final String name = matcher.group(2);
99  			final String[] names = name.split(",", 2);
100 			if (names.length == 1) {
101 				actionFullName = StringUtil.replace(actionFullName, matcher
102 						.group(1), "([a-zA-Z0-9]+)");
103 				uriParameterNames.add(matcher.group(2));
104 			} else {
105 				actionFullName = StringUtil.replace(actionFullName, matcher
106 						.group(1), "(" + names[1] + ")");
107 				uriParameterNames.add(names[0]);
108 			}
109 		}
110 
111 		final String forwardPath = this.fromActionClassToPath(actionClass,
112 				method);
113 
114 		final RoutingInfo routingInfo = new RoutingInfo(actionClass, method,
115 				uriParameterNames, forwardPath, requestMethods);
116 		final Pattern pattern = Pattern.compile("^" + actionFullName + "$");
117 
118 		patternToRoutingInfoMap.put(pattern, routingInfo);
119 		if (logger.isDebugEnabled()) {
120 			logger.log("DCUB0007", new Object[] { actionFullName, method,
121 					uriParameterNames });
122 		}
123 	}

126 			final Class<? extends Action> actionClass, final Method method) {
127 		final String componentName = namingConvention
128 				.fromClassNameToComponentName(actionClass.getCanonicalName());
129 		final StringBuilder builder = new StringBuilder(100);
130 		builder.append('/');
131 		builder.append(componentName.substring(
132 				0,
133 				componentName.length()
134 						- namingConvention.getActionSuffix().length())
135 				.replaceAll("_", "/"));

163 	private ForwardInfo findForwardInfo(final String path,
164 			final Map<Pattern, RoutingInfo> routingPatterns) {
165 		final Map<String, String> uriParams = new HashMap<String, String>();
166 		for (final Entry<Pattern, RoutingInfo> entry : routingPatterns
167 				.entrySet()) {
168 			final Matcher matcher = entry.getKey().matcher(path);
169 			if (matcher.find()) {
170 				final RoutingInfo routingInfo = entry.getValue();
171 				final HttpServletRequest request = SingletonS2Container
172 						.getComponent(HttpServletRequest.class);
173 				if (routingInfo.isAcceptable(request)) {
174 					for (int i = 1; i < matcher.groupCount() + 1; i++) {
175 						final String name = routingInfo.getUriParameterNames()
176 								.get(i - 1);
177 						final String value = matcher.group(i);
178 						uriParams.put(name, value);
179 					}
180 					final ForwardInfoImpl forwardInfo = new ForwardInfoImpl(
181 							routingInfo, uriParams);
182 					return forwardInfo;
183 				}
184 			}
185 		}
186 		return null;
187 	}

217 			final StringBuilder builder = new StringBuilder(100);

222 				final String encoding = PathResolverImpl.this.uriEncoding;
223 				if (!StringUtil.isEmpty(encoding)) {

273 		public void processClass(final String packageName,
274 				final String shortClassName) {
275 			if (shortClassName.indexOf('$') != -1) {
276 				return;
277 			}
278 			final String className = ClassUtil.concatName(packageName,
279 					shortClassName);
280 			if (!namingConvention.isTargetClassName(className)) {
281 				return;
282 			}
283 			if (!className.endsWith(namingConvention.getActionSuffix())) {
284 				return;
285 			}
286 			final Class<? extends Action> clazz = classForName(className);
287 			if (namingConvention.isSkipClass(clazz)) {
288 				return;
289 			}
290 
291 			for (final Method method : clazz.getMethods()) {
292 				if (CubbyUtils.isActionMethod(method)) {
293 					final String actionFullName = CubbyUtils.getActionUrl(
294 							clazz, method);
295 					final Url.RequestMethod[] acceptableRequestMethods = CubbyUtils
296 							.getAcceptableRequestMethods(clazz, method);
297 					add(actionFullName, clazz, method, routingPatterns,
298 							acceptableRequestMethods);
299 				}
300 			}
301 		}
