44  /**
45   * リクエストされたURLを適切なアクションに振り分けるフィルタ。
46   * <p>
47   * {@link PathResolver} によって {@link InternalForwardInfo}
48   * を抽出し、そこに保持された情報をもとにフォワードします。
49   * </p>
50   * 
51   * @author baba
52   */
53  public class RequestRoutingFilter implements Filter {
54  
55  	/** ロガー */
56  	private static final Logger logger = Logger
57  			.getLogger(RequestRoutingFilter.class);
58  
59  	/** ルーティングの対象外とするパスの初期パラメータ名 */
60  	public static final String IGNORE_PATH_PATTERN = "ignorePathPattern";
61  
62  	/** ルーター */
63  	private static final Router router = new Router();
64  
65  	/** ルーティングの対象外とするパスの正規表現パターンのリスト */
66  	private final List<Pattern> ignorePathPatterns = new ArrayList<Pattern>();
67  
68  	/**
69  	 * このフィルタを初期化します。
70  	 * <p>
71  	 * 使用可能な初期化パラメータ <table> <thead>
72  	 * <th> 初期化パラメータ名 </th>
73  	 * <th> 初期化パラメータの値 </th>
74  	 * <th> 例 </th>
75  	 * </thead> <thead>
76  	 * <tr>
77  	 * <td>{@link #IGNORE_PATH_PATTERN}</td>
78  	 * <td>ルーティングの対象外とするパスの正規表現をカンマ区切りで指定します。 HotDeploy
79  	 * 時のパフォーマンスにも影響するので、画像やスクリプトを特定のディレクトリに
80  	 * 格納していてアクションを実行するパスと明確に区別できる場合はできる限り指定するようにしてください。</td>
81  	 * <td>
82  	 * 
83  	 * <pre>
84  	 * &lt;param-name&gt;ignorePathPattern&amp;lt/param-name&gt;
85  	 * &lt;param-value&gt;/img/.*,/js/.*&lt;param-name&gt;
86  	 * </pre>
87  	 * 
88  	 * この例では /img と /js 以下のパスをルーティングの対象外にします。 </td>
89  	 * </tr>
90  	 * </thead>
91  	 * </p>
92  	 * 
93  	 * @param config
94  	 *            Filter 設定のためのオブジェクト
95  	 * @exception ServletException
96  	 *                初期化処理で例外が発生した場合
97  	 */
98  	public void init(final FilterConfig config) throws ServletException {
99  		final String ignorePathPatternString = config
100 				.getInitParameter(IGNORE_PATH_PATTERN);
101 		if (!StringUtil.isEmpty(ignorePathPatternString)) {
102 
103 			for (final StringTokenizer tokenizer = new StringTokenizer(
104 					ignorePathPatternString, ","); tokenizer.hasMoreTokens();) {
105 				final String token = tokenizer.nextToken();
106 				final Pattern pattern = Pattern.compile(token);
107 				ignorePathPatterns.add(pattern);
108 			}
109 		}
110 	}

116 	}

118 	/**
119 	 * フィルタリングを行います。
120 	 * <p>
121 	 * リクエストされた URI に対応する内部フォワード用のパスが {@link Router} から取得できた場合は、
122 	 * その内部フォワード用のパスへフォワードします。フォワード先は {@link CubbyFilter} が処理することを期待します。
123 	 * マッチするパターンがなかった場合はフィルタチェインで次のフィルタに処理を移譲します。
124 	 * </p>
125 	 * 
126 	 * @param request
127 	 *            リクエスト
128 	 * @param response
129 	 *            レスポンス
130 	 * @param chain
131 	 *            フィルタチェイン
132 	 * @throws IOException
133 	 *             リクエストディスパッチャやフィルタチェインで例外が発生した場合
134 	 * @throws ServletException
135 	 *             リクエストディスパッチャやフィルタチェインで例外が発生した場合
136 	 * @see Router#routing(HttpServletRequest, HttpServletResponse, List)
137 	 * @see CubbyFilter
138 	 */
139 	public void doFilter(final ServletRequest request,
140 			final ServletResponse response, final FilterChain chain)
141 			throws IOException, ServletException {
142 
143 		final InternalForwardInfo internalForwardInfo = router.routing(
144 				(HttpServletRequest) request, (HttpServletResponse) response,
145 				ignorePathPatterns);
146 		if (internalForwardInfo != null) {
147 			final String internalForwardPath = internalForwardInfo
148 					.getInternalForwardPath();
149 			if (logger.isDebugEnabled()) {
150 				logger.log("DCUB0001", new Object[] { internalForwardPath });
151 			}
152 			final RequestDispatcher requestDispatcher = request
153 					.getRequestDispatcher(internalForwardPath);
154 			requestDispatcher.forward(request, response);
155 		} else {
156 			chain.doFilter(request, response);
157 		}
158 	}

160 	/**
161 	 * ルーター。
162 	 * 
163 	 * @author baba
164 	 */
165 	public static class Router {
166 
167 		/** 空の対象外パターンのリスト */
168 		private static final List<Pattern> EMPTY_IGNORE_PATH_PATTERNS = Collections
169 				.emptyList();
170 
171 		/**
172 		 * 対象外パターンを指定せずにルーティング処理を行い、内部フォワード情報を返します。
173 		 * 
174 		 * @param request
175 		 *            リクエスト
176 		 * @param response
177 		 *            レスポンス
178 		 * @return リクエスト URI に対応する内部フォワード情報、URI に対応する内部フォワード情報がない場合は
179 		 *         <code>null</code>
180 		 * @see #routing(HttpServletRequest, HttpServletResponse, List)
181 		 */
182 		public InternalForwardInfo routing(final HttpServletRequest request,
183 				final HttpServletResponse response) {
184 			return routing(request, response, EMPTY_IGNORE_PATH_PATTERNS);
185 		}
186 
187 		/**
188 		 * リクエストのルーティング処理を行い、内部フォワード情報を返します。
189 		 * <p>
190 		 * このメソッドはリクエスト URI とメソッドに対応するフォワード情報を{@link PathResolver} によって決定します。
191 		 * </p>
192 		 * 
193 		 * @param request
194 		 *            リクエスト
195 		 * @param response
196 		 *            レスポンス
197 		 * @param ignorePathPatterns
198 		 *            対象外とするパスのパターン
199 		 * @return リクエストに対応する内部フォワード情報、URI とリクエストメソッドに対応する内部フォワード情報がない場合や URI
200 		 *         が対象外とするパスのパターンにマッチする場合は <code>null</code>
201 		 * @see PathResolver#getInternalForwardInfo(String, String)
202 		 * @see org.seasar.cubby.action.Path
203 		 * @see org.seasar.cubby.action.Accept
204 		 */
205 		public InternalForwardInfo routing(final HttpServletRequest request,
206 				final HttpServletResponse response,
207 				List<Pattern> ignorePathPatterns) {
208 			final String path = CubbyUtils.getPath(request);
209 			if (logger.isDebugEnabled()) {
210 				logger.log("DCUB0006", new Object[] { path });
211 			}
212 
213 			if (isIgnorePath(path, ignorePathPatterns)) {
214 				return null;
215 			}
216 
217 			final S2Container container = SingletonS2ContainerFactory
218 					.getContainer();
219 			final PathResolver pathResolver = (PathResolver) container
220 					.getComponent(PathResolver.class);
221 
222 			final InternalForwardInfo internalForwardInfo = pathResolver
223 					.getInternalForwardInfo(path, request.getMethod());
224 			return internalForwardInfo;
225 		}
226 
227 		/**
228 		 * 指定された path が ignorePathPatterns にマッチするかを示します。
229 		 * 
230 		 * @param path
231 		 *            パス
232 		 * @param ignorePathPatterns
233 		 *            対象外パターンのリスト
234 		 * @return path が ignorePathPatterns にマッチする場合は <code>true</code>、そうでない場合は
235 		 *         <code>false</code>
236 		 */
237 		private boolean isIgnorePath(final String path,
238 				List<Pattern> ignorePathPatterns) {
239 			for (final Pattern pattern : ignorePathPatterns) {
240 				final Matcher matcher = pattern.matcher(path);
241 				if (matcher.matches()) {
242 					return true;
243 				}
244 			}
245 			return false;
246 		}
247 
248 	}
