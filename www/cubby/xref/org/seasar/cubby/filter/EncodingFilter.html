19  import java.io.UnsupportedEncodingException;
20  import java.nio.charset.Charset;

31  /**
32   * リクエストのエンコーディングを設定するためのフィルタです。
33   * <p>
34   * 初期化パラメータ {@value #ENCODING}、{@value #FORCE_ENCODING} でリクエストの文字エンコーディングを指定します。
35   * </p>
36   * <p>
37   * 初期化パラメータ {@value #URI_ENCODING}、{@value #URI_BYTES_ENCODING} で
38   * {@link HttpServletRequest#getServletPath()}、
39   * {@link HttpServletRequest#getPathInfo()} で取得できるパスのエンコーディングを指定します。
40   * </p>
41   * <p>
42   * <table>
43   * <thead>
44   * <tr>
45   * <th>param-name</th>
46   * <th>param-value</th>
47   * </tr>
48   * <tbody>
49   * <tr>
50   * <td>{@value #ENCODING}</td>
51   * <td>リクエストのエンコーディングを指定します。リクエストのエンコーディングが <code>null</code> か、
52   * {@value #FORCE_ENCODING} に <code>true</code>
53   * が指定された場合はこのエンコーディングがリクエストに設定されます。</td>
54   * </tr>
55   * <tr>
56   * <td>{@value #FORCE_ENCODING}</td>
57   * <td><code>true</code> を指定した場合は、リクエストにエンコーディングが設定されていても {@value #ENCODING}
58   * で上書きします。
59   * </tr>
60   * <tr>
61   * <td>{@value #URI_ENCODING}</td>
62   * <td>URI のエンコーディングを指定します。</td>
63   * </tr>
64   * <tr>
65   * <td>{@value #URI_BYTES_ENCODING}</td>
66   * <td>URI をバイト配列として取得する際のエンコーディングを指定します。</td>
67   * </tr>
68   * </tbody>
69   * </table>
70   * <caption>初期化パラメータ</caption>
71   * </p>
72   * 
73   * @author baba
74   * @since 1.1.1
75   */
76  public class EncodingFilter implements Filter {
77  
78  	private static final String ALREADY_FILTERED_ATTRIBUTE_NAME = EncodingFilter.class
79  			.getName()
80  			+ ".FILTERED";
81  
82  	/** エンコーディングのキー。 */
83  	private static final String ENCODING = "encoding";
84  
85  	/** 強制エンコーディング設定のキー。 */
86  	private static final String FORCE_ENCODING = "forceEncoding";
87  
88  	/** URI エンコーディングのキー。 */
89  	private static final String URI_ENCODING = "URIEncoding";
90  
91  	/** URI バイト列のエンコーディングのキー。 */
92  	private static final String URI_BYTES_ENCODING = "URIBytesEncoding";
93  
94  	/** URI バイト列のエンコーディングのデフォルト値。 */
95  	private static final String DEFAULT_URI_BYTE_ENCODING = "ISO-8859-1";
96  
97  	/** エンコーディング。 */
98  	private String encoding;
99  
100 	/** 強制エンコーディング設定。 */
101 	private boolean forceEncoding = false;
102 
103 	/** URI エンコーディング。 */
104 	private String uriEncoding;
105 
106 	/** URI バイト列のエンコーディング。 */
107 	private String uriBytesEncoding;
108 
109 	/**
110 	 * {@inheritDoc}
111 	 */
112 	public void init(final FilterConfig config) throws ServletException {
113 		encoding = config.getInitParameter(ENCODING);
114 		try {
115 			validateEncoding(encoding);
116 		} catch (final UnsupportedEncodingException e) {
117 			throw new ServletException(e);
118 		}
119 
120 		final String forceEncodingString = config
121 				.getInitParameter(FORCE_ENCODING);
122 		if (forceEncodingString != null) {
123 			forceEncoding = Boolean.parseBoolean(forceEncodingString);
124 		}
125 
126 		uriEncoding = config.getInitParameter(URI_ENCODING);
127 		try {
128 			validateEncoding(uriEncoding);
129 		} catch (final UnsupportedEncodingException e) {
130 			throw new ServletException(e);
131 		}
132 
133 		uriBytesEncoding = config.getInitParameter(URI_BYTES_ENCODING);
134 		if (uriBytesEncoding == null) {
135 			uriBytesEncoding = DEFAULT_URI_BYTE_ENCODING;
136 		}
137 		try {
138 			validateEncoding(uriBytesEncoding);
139 		} catch (final UnsupportedEncodingException e) {
140 			throw new ServletException(e);
141 		}
142 	}
143 
144 	/**
145 	 * 指定されたエンコーディングがサポートされているか検査します。
146 	 * 
147 	 * @param encoding
148 	 *            エンコーディング
149 	 * @throws UnsupportedEncodingException
150 	 *             指定されたエンコーディングがサポートされていない場合
151 	 */
152 	private void validateEncoding(final String encoding)
153 			throws UnsupportedEncodingException {
154 		if (encoding != null) {
155 			if (!Charset.isSupported(encoding)) {
156 				throw new UnsupportedEncodingException(encoding);
157 			}
158 		}
159 	}
160 
161 	/**
162 	 * {@inheritDoc}
163 	 */
164 	public void destroy() {
165 	}
166 
167 	/**
168 	 * {@inheritDoc}
169 	 */
170 	public void doFilter(final ServletRequest request,
171 			final ServletResponse response, final FilterChain chain)
172 			throws IOException, ServletException {
173 		if (request.getAttribute(ALREADY_FILTERED_ATTRIBUTE_NAME) == null) {
174 			request.setAttribute(ALREADY_FILTERED_ATTRIBUTE_NAME, Boolean.TRUE);
175 			if (request.getCharacterEncoding() == null || forceEncoding) {
176 				request.setCharacterEncoding(encoding);
177 			}
178 			if (uriEncoding == null) {
179 				chain.doFilter(request, response);
180 			} else {
181 				final ServletRequest wrapper = new EncodingHttpServletRequestWrapper(
182 						(HttpServletRequest) request, null, null);
183 				chain.doFilter(wrapper, response);
184 			}
185 			request.removeAttribute(ALREADY_FILTERED_ATTRIBUTE_NAME);
186 		} else {
187 			chain.doFilter(request, response);
188 		}
189 	}
190 
191 	/**
192 	 * 適切なエンコーディングで処理するための {@link HttpServletRequestWrapper} です。
193 	 * 
194 	 * @author baba
195 	 * @since 1.1.1
196 	 */
197 	private class EncodingHttpServletRequestWrapper extends
198 			HttpServletRequestWrapper {
199 
200 		/** URI エンコーディング。 */
201 		private final String uriEncoding;
202 
203 		/** URI バイト列のエンコーディング。 */
204 		private final String uriBytesEncoding;
205 
206 		/**
207 		 * 指定されたリクエストをラップします。
208 		 * 
209 		 * @param request
210 		 *            リクエスト
211 		 * @param uriEncoding
212 		 *            URI エンコーディング
213 		 * @param uriBytesEncoding
214 		 *            URI バイト列のエンコーディング。
215 		 * @throws IOException
216 		 *             スーパクラスのコンストラクタで例外が発生した場合
217 		 */
218 		public EncodingHttpServletRequestWrapper(
219 				final HttpServletRequest request, final String uriEncoding,
220 				final String uriBytesEncoding) throws IOException {
221 			super(request);
222 			this.uriEncoding = uriEncoding;
223 			this.uriBytesEncoding = uriBytesEncoding;
224 		}
225 
226 		@Override
227 		public String getServletPath() {
228 			return rebuild(super.getServletPath(), uriEncoding,
229 					uriBytesEncoding);
230 		}
231 
232 		@Override
233 		public String getPathInfo() {
234 			return rebuild(super.getPathInfo(), uriEncoding, uriBytesEncoding);
235 		}
236 
237 		/**
238 		 * 指定された文字列を一度 <code>bytesEncoding</code> でバイト配列に戻し、
239 		 * <code>encoding</code> で文字列を再構築します。
240 		 * 
241 		 * @param str
242 		 *            文字列
243 		 * @param encoding
244 		 *            エンコーディング
245 		 * @param bytesEncoding
246 		 *            バイト配列に戻す時のエンコーディング
247 		 * @return 再構築された文字列
248 		 */
249 		private String rebuild(final String str, final String encoding,
250 				final String bytesEncoding) {
251 			if (str == null || encoding == null) {
252 				return str;
253 			}
254 			try {
255 				return new String(str.getBytes(bytesEncoding), encoding);
256 			} catch (final UnsupportedEncodingException e) {
257 				throw new IllegalStateException(e);
258 			}
259 		}
260 
261 	}
