26   * inputを出力するタグ。
27   * <p>
28   * 入力検証にエラーがある場合、class属性に「fieldError」を追加します。 なおこのタグはtype属性により挙動が変わります。
29   * </p>
30   * <ul>

32   * value値をvalue属性の値として出力します。フォームオブジェクトの値とvalueが一致した場合checked="checked"を出力します。</li>
33   * <li>その他 - value値をvalue属性の値として出力します。</li>
34   * </ul>

37   * @author baba
38   */

74  	@Override
75  	public void doTag() throws JspException, IOException {
76  		final JspContext context = this.getJspContext();
77  		final JspWriter out = context.getOut();
78  		final ActionErrors errors = errors(context);
79  		final Map<String, Object> dyn = this.getDynamicAttribute();
80  		final Map<String, String[]> outputValues = outputValues(context);
81  
82  		if (this.index == null) {
83  			if (!errors.getFields().get(this.name).isEmpty()) {
84  				addClassName(dyn, "fieldError");
85  			}
86  		} else {
87  			if (!errors.getIndexedFields().get(this.name).get(index).isEmpty()) {
88  				addClassName(dyn, "fieldError");
89  			}
90  		}
91  
92  		if (ArrayUtil.contains(MULTIPLE_VALUE_TYPES, this.type)) {
93  			if (this.value == null) {
94  				throw new JspTagException(MessageFormatter.getMessage(
95  						"ECUB1003", new Object[] {
96  								Arrays.deepToString(MULTIPLE_VALUE_TYPES),
97  								"value" }));
98  			}
99  			final Object[] values = multipleFormValues(context, outputValues,
100 					this.name, this.checkedValue);
101 
102 			out.write("<input type=\"");
103 			out.write(type);
104 			out.write("\" name=\"");
105 			out.write(this.name);
106 			out.write("\" value=\"");
107 			out.write(CubbyFunctions.out(this.value));// TODO
108 			out.write("\" ");
109 			out.write(toAttr(dyn));
110 			out.write(" ");
111 			out.write(checked(toString(this.value), values));
112 			out.write("/>\n");
113 		} else {
114 			final Object value = formValue(context, outputValues, this.name,
115 					this.index, this.value);
116 
117 			out.write("<input type=\"");
118 			out.write(type);
119 			out.write("\" name=\"");
120 			out.write(this.name);
121 			out.write("\" value=\"");
122 			out.write(CubbyFunctions.out(toString(value)));
123 			out.write("\" ");
124 			out.write(toAttr(dyn));
125 			out.write("/>\n");
126 		}
127 	}
