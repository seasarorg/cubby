61  			attribute = outputValues;

66  			final BeanDesc beanDesc = BeanDescFactory.getBeanDesc(concreteClass);
67  			if (beanDesc.hasPropertyDesc(name)) {

69  				if (propertyDesc.isReadable()) {

73  				}
