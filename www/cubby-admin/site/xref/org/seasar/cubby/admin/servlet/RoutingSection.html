38  class RoutingSection implements Section {
39  
40  	private HttpServletRequest request;
41  
42  	private PrintWriter out;
43  
44  	private Locale locale;
45  
46  	private PathResolver pathResolver;
47  
48  	private Iterator<String> rowClasses = new LoopingIterator<String>(Arrays
49  			.asList(new String[] { "odd", "even" }));

53  	public RoutingSection(HttpServletRequest request,
54  			HttpServletResponse response) throws IOException {
55  		this.request = request;
56  		this.out = response.getWriter();
57  		this.locale = request.getLocale();
58  		this.pathResolver = (PathResolverImpl) SingletonS2Container
59  				.getComponent(PathResolver.class);
60  		this.messages = new Messages(locale);
61  	}
62  
63  	public void print() {
64  		final String path = request.getParameter("path");
65  
66  		out.println("<form name=\"pathSerachForm\">");
67  
68  		out.print("<label for=\"path\">");
69  		out.print(messages.getString("lbl.path"));
70  		out.print("</label>");
71  		out.print("<input type=\"text\" name=\"path\" size=\"40\" value=\"");
72  		out.print(escapeHtml(path));
73  		out.println("\" />");
74  
75  		out.print("<input type=\"submit\" value=\"");
76  		out.print(messages.getString("lbl.test"));
77  		out.println("\">");
78  
79  		out.print("<input type=\"button\" value=\"");
80  		out.print(messages.getString("lbl.clear"));
81  		out
82  				.println("\" onclick=\"pathSerachForm.path.value='';pathSerachForm.submit();\">");
83  
84  		out.println("</form>");
85  
86  		out.println("<table>");
87  		out.println("<thead>");
88  		out.println(th(messages.getString("lbl.no")));
89  		out.println(th(messages.getString("lbl.regexp")));
90  		out.println(th(messages.getString("lbl.requestMethod")));
91  		out.println(th(messages.getString("lbl.actionMethod")));
92  		out.println(th(messages.getString("lbl.pathParams")));
93  		out.println(th(messages.getString("lbl.priority")));
94  		out.println("</thead>");
95  		out.println("<tbody>");
96  		int no = 1;
97  		for (final Routing routing : pathResolver.getRoutings()) {
98  			final boolean write;
99  			final Map<String, String> pathParameters;
100 			if (isEmpty(path)) {
101 				write = true;
102 				pathParameters = null;
103 			} else {
104 				final Matcher matcher = routing.getPattern().matcher(path);
105 				if (matcher.matches()) {
106 					write = true;
107 					pathParameters = new HashMap<String, String>();
108 					for (int i = 0; i < matcher.groupCount(); i++) {
109 						final String name = routing.getUriParameterNames().get(
110 								i);
111 						final String value = matcher.group(i + 1);
112 						pathParameters.put(name, value);
113 					}
114 				} else {
115 					write = false;
116 					pathParameters = null;
117 				}
118 			}
119 			if (write) {
120 				out.print("<tr class=\"");
121 				out.print(rowClasses.next());
122 				out.println("\">");
123 				out.println(td(no));
124 				out.println(td(routing.getPattern().pattern()));
125 				out.println(td(routing.getRequestMethod()));
126 				out.println(td(routing.getActionClass().getName() + "#"
127 						+ routing.getMethod().getName()));
128 				if (pathParameters == null) {
129 					out.println(td(routing.getUriParameterNames()));
130 				} else {
131 					out.println(td(pathParameters));
132 				}
133 				out.println(td(routing.getPriority()));
134 				out.println("</tr>");
135 			}
136 			no++;
137 		}
138 		out.println("</tbody>");
139 		out.println("</table>");
140 	}
